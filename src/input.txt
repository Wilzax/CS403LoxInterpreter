class Circle{
  init(radius){
    this.radius = radius;
    this.area = this.area();
    return;
  }

  area() {
    return 3.141592654 * this.radius * this.radius;
  }
}

var circle = Circle(4);
print "------------------------------------";
print circle.radius; //Prints 4
print circle.area; //Prints ~50.2655
print "------------------------------------";
